#include <stdio.h>

/* Include the mruby header */
#include <mruby.h>
#include <mruby/compile.h>
#include <mruby/array.h>

int main(void)
{
  mrb_state *mrb = mrb_open();

  FILE *file = fopen("example.rb", "r");
  mrb_load_file(mrb, file);
  fclose(file);
}
